# tuseday

localStorage - profileElement
{
    title: 블로그 제목
    content: 프로필 내용
    user: 사용자 (== userName일 때 show)
    src: 이미지 주소
}

localStorage - articleElement
{
    title: 제목
    content: 내용
    author: 작성자 (== userName일 때 show)
    src: 이미지 주소
    num: 일련번호
}

localStorage - userName
{
    userName: 보고자 하는 블로그의 작성자
    loggedinUser: 현재 로그인한 사용자
    loginSuccess: 로그인 유무
}
